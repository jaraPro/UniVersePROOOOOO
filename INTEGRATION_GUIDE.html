Документация по интеграции Auth.js

Данная библиотека предоставляет единую систему аутентификации для всех страниц сайта UniVerse Pro. Она обрабатывает регистрацию, вход, выход и хранит данные в localStorage.

Готово к использованию! Просто добавьте <script src="auth.js"></script> перед

Быстрый старт

1. Подключить скрипт:

<link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/font-awesome/6.4.0/css/all.min.css">
<script src="auth.js"></script>

2. Добавить кнопку входа:

<button onclick="universeAuth.openModal()">
  <i class="fas fa-sign-in-alt"></i> Войти
</button>

3. Проверить аутентификацию:

// В JavaScript
if (universeAuth.isAuthenticated()) {
  console.log("Пользователь вошел:", universeAuth.currentUser);
} else {
  console.log("Пользователь не авторизован");
}

API Справка

Свойства:

  • universeAuth.currentUser - Объект текущего пользователя или null
  • universeAuth.isAuthenticated() - Возвращает true/false

Методы:

  • universeAuth.openModal() - Открыть модальное окно регистрации
  • universeAuth.closeModal() - Закрыть модальное окно
  • universeAuth.switchTab(tab) - Переключить вкладку ('login' или 'register')
  • universeAuth.register(name, email, phone, password) - Регистрация
  • universeAuth.login(email, password) - Вход
  • universeAuth.logout() - Выход
  • universeAuth.showError(message) - Показать ошибку
  • universeAuth.showSuccess(message) - Показать успех

Примеры использования

Пример 1: Защита контента

<div id="content"></div>

<script src="auth.js"></script>
<script>
  window.addEventListener('load', () => {
    if (!universeAuth.isAuthenticated()) {
      document.getElementById('content').innerHTML = 
        '<p>Пожалуйста, <button onclick="universeAuth.openModal()">войдите</button></p>';
    } else {
      console.log("Добро пожаловать,", universeAuth.currentUser.name);
    }
  });
</script>

Пример 2: Показать данные пользователя

<div id="userInfo"></div>

<script>
  if (universeAuth.isAuthenticated()) {
    const user = universeAuth.currentUser;
    document.getElementById('userInfo').innerHTML = `
      <p><strong>Привет, ${user.name}!</strong></p>
      <p>Email: ${user.email}</p>
      <p>Дата регистрации: ${new Date(user.createdAt).toLocaleDateString('ru-RU')}</p>
    `;
  }
</script>

Пример 3: Проверка перед отправкой

function submitForm(data) {
  if (!universeAuth.isAuthenticated()) {
    alert("Сначала войдите в систему!");
    universeAuth.openModal();
    return;
  }
  
  // Отправить данные с информацией пользователя
  data.userId = universeAuth.currentUser.id;
  // ... отправка формы ...
}

Пример 4: Условный рендер навигации

<script>
  window.addEventListener('load', () => {
    const isAuth = universeAuth.isAuthenticated();
    const loginBtn = document.getElementById('loginBtn');
    const userPanel = document.getElementById('userPanel');
    
    if (isAuth) {
      loginBtn.style.display = 'none';
      userPanel.style.display = 'flex';
      document.getElementById('userName').textContent = universeAuth.currentUser.name;
    } else {
      loginBtn.style.display = 'flex';
      userPanel.style.display = 'none';
    }
  });
</script>

Структура данных

Объект пользователя:

{
  id: 1234567890,              // Уникальный ID (timestamp)
  name: "Иван Иванов",         // Имя
  email: "ivan@example.com",   // Email (уникальный)
  phone: "+7 (777) 123-45-67", // Телефон (опционально)
  password: "password123",     // Пароль (минимум 8 символов)
  bio: "...",                  // Описание профиля
  city: "Алматы",              // Город
  picture: null,               // Аватар (для расширения)
  provider: "local",           // "local" или "google"
  isEmailVerified: false,      // Email подтвержден?
  isPhoneVerified: false,      // Phone подтвержден?
  createdAt: "2026-04-01T...", // ISO дата регистрации
  favorites: []                // Массив ID избранных
}
Внимание! Пароли хранятся в открытом виде в localStorage. Для production нужно использовать backend с хешированием!

Если добавить на другие страницы:

Просто добавьте перед :

<script src="auth.js"></script>

И кнопку:

<button onclick="universeAuth.openModal()">Вход</button>

Правила валидации

  • Имя: минимум 2 символа
  • Email: должен содержать @ и быть уникальным
  • Пароль: минимум 8 символов
  • Телефон: любой формат (опционально)

Сообщения об ошибках:

  • "Имя должно содержать минимум 2 символа"
  • "Некорректный email"
  • "Пароль должен содержать минимум 8 символов"
  • "Email уже зарегистрирован"
  • "Неверный email или пароль"
  • "Пароли не совпадают"

Best Practices

  • Всегда проверяйте universeAuth.isAuthenticated() перед показом private контента
  • Сохраняйте данные пользователя в universeAuth.currentUser
  • Не забывайте добавить Font Awesome для иконок
  • Вызывайте location.reload() после входа/выхода если нужна синхронизация
  • Используйте localStorage ключи: 'universe_users', 'universe_currentUser'
  • Тестируйте в разных браузерах и мобильных устройствах
